# etl_pipeline.py — extract()
def extract() -> dict[str, pd.DataFrame]:
    customers = pd.read_csv("data/olist_customers_dataset.csv")

    orders = pd.read_csv("data/olist_orders_dataset.csv",
                          parse_dates=["order_purchase_timestamp",
                                       "order_approved_at",
                                       "order_delivered_customer_date"])

    payments = pd.read_csv("data/olist_order_payments_dataset.csv")

    return {"customers": customers, "orders": orders, "payments": payments}

The extract phase reads three CSV files. In production you would swap pd.read_csv() for database queries or API calls — the rest of the pipeline stays unchanged.

Phase 2

Transform

1

Standardise column names & types

Lowercase, strip whitespace, title-case city names, upper-case state codes.

2

Deduplicate customers

Drop duplicate rows on customer_unique_id. The raw dataset contains ~2 % duplicates.

3

Derive order metrics per customer

Join payments → orders, filter to delivered orders only, aggregate: total orders, total revenue, average order value, average delivery days.

4

Customer value segmentation

VIP — ≥3 orders AND ≥R$500 revenue · LOYAL — ≥2 orders OR ≥R$200 · NEW — everyone else.

5

Anonymise customer IDs

Replace raw IDs with a truncated SHA-256 hash. Preserves referential integrity while protecting PII.

6

Add ETL audit columns

Stamp every row with etl_loaded_at (UTC) and etl_pipeline_ver for lineage tracking.

# Segment logic (applied with df.apply)
def segment(row) -> str:
    if   row["total_orders"] >= 3 and row["total_revenue"] >= 500:
        return "VIP"
    elif row["total_orders"] >= 2 or  row["total_revenue"] >= 200:
        return "LOYAL"
    else:
        return "NEW"

# SHA-256 anonymisation
def _mask_id(raw_id: str) -> str:
    return hashlib.sha256(raw_id.encode()).hexdigest()[:16]

# Aggregate delivered orders per customer
customer_stats = (

def load(df: pd.DataFrame) -> None:
    conn = snowflake.connector.connect(**SNOWFLAKE_CONFIG)
    cur  = conn.cursor()

    try:
        cur.execute(DDL_CREATE_TABLE)   # CREATE TABLE IF NOT EXISTS
        cur.execute("TRUNCATE TABLE IF EXISTS DIM_CUSTOMERS;")

        # Bulk upload via Snowflake's optimised write_pandas
        success, nchunks, nrows, _ = write_pandas(
            conn,
            df,
            table_name="DIM_CUSTOMERS",
            quote_identifiers=False,
        )
    finally:
        cur.close()
        conn.close()
python

The load uses a full-refresh strategy (TRUNCATE + INSERT). For large datasets, replace with a MERGE statement keyed on customer_key to enable incremental loads without re-processing everything.

Orchestrator

run_pipeline()

def run_pipeline() -> None:
    start = datetime.utcnow()

    raw   = extract()           # Phase 1
    clean = transform(raw)      # Phase 2
    load(clean)                  # Phase 3

    elapsed = (datetime.utcnow() - start).total_seconds()
    log.info(f"PIPELINE COMPLETE ({elapsed:.1f}s)")

if __name__ == "__main__":
    run_pipeline()
python

Each phase is a pure function with a clear contract: extract returns raw DataFrames, transform returns a clean DataFrame, load takes a clean DataFrame. This makes unit-testing and swapping components trivial.
